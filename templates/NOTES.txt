Clair has been deployed in COMBO mode (all-in-one process).

  Release:   {{ .Release.Name }}
  Namespace: {{ .Release.Namespace }}
  Image:     {{ .Values.image.repository }}:{{ .Values.image.tag | default .Chart.AppVersion }}

Get the Clair API address:
{{- if .Values.combo.ingress.enabled }}
{{- range .Values.combo.ingress.hosts }}
  http{{ if $.Values.combo.ingress.tls }}s{{ end }}://{{ .host }}
{{- end }}
{{- else if eq .Values.combo.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "clair.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "clair.fullname" . }} {{ .Values.combo.service.port }}:{{ .Values.combo.service.port }}
  Then access: http://127.0.0.1:{{ .Values.combo.service.port }}
{{- end }}

    psql -U {{ .Values.database.user }} {{ .Values.database.name }}

  External database: {{ .Values.database.externalConnString | default "(configure database.externalConnString)" }}
